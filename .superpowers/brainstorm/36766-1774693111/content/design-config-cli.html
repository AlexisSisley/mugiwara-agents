Section 6 — Configuration, Theming & CLI

TOML config, thèmes, fonts, et CLI scriptable

⚙️ config.toml — Configuration principale
# ~/.cmux/config.toml

[daemon]
auto_start = true # Lance le daemon au démarrage du GUI
detach_timeout = "30m" # Durée avant hibernation sans client
scrollback_lines = 10000 # Lignes de scrollback par pane
save_interval_ms = 500 # Debounce pour session.json
scrollback_flush_s = 30 # Flush scrollback sur disque
log_level = "info" # trace | debug | info | warn | error

[terminal]
shell = "pwsh.exe" # Shell par défaut (auto-detect si absent)
env = { TERM = "xterm-256color", COLORTERM = "truecolor" }
cursor_style = "block" # block | underline | bar
cursor_blink = true
bell = "visual" # none | visual | audible

[gpu]
backend = "auto" # auto | vulkan | dx12 | metal
vsync = true
frame_rate_cap = 0 # 0 = vsync, sinon cap en fps

[claude]
detect = true # Active la détection Claude Code
process_poll_ms = 2000 # Intervalle polling process tree
pattern_analysis = true # Niveau 3 : output pattern analysis
notifications = true # Affiche les popups de notification
border_glow = true # Bordure colorée selon le statut

[ui]
sidebar = true # Affiche la sidebar
sidebar_width = 220 # Largeur en pixels
statusbar = true # Affiche la status bar
tab_bar_position = "sidebar" # sidebar | top | hidden
notification_duration_s = 5
padding = { x = 8, y = 4 } # Padding intérieur des panes

🟣 cmux CLI — Commandes complètes
Commande Description Exemple
cmux start Lance le daemon en background cmux start --log-level debug
cmux attach Ouvre le GUI et connecte au daemon cmux attach --workspace 2
cmux ls Liste workspaces/tabs/panes cmux ls --format json
cmux new-workspace Crée un workspace cmux new-workspace --name "api" --cwd ~/projects/api
cmux new-tab Crée un tab dans le workspace actif cmux new-tab --name "tests"
cmux split Split le pane actif cmux split --direction right --ratio 0.6
cmux send-keys Envoie des touches à un pane cmux send-keys --pane 3 "cargo test\n"
cmux focus Focus un workspace/tab/pane cmux focus --workspace 1 --tab 2 --pane 1
cmux events Stream d'events (Claude, commandes...) cmux events --follow --filter claude
cmux snapshot Capture l'état d'un pane cmux snapshot --pane 2 --output screen.txt
cmux kill Arrête le daemon proprement cmux kill --save
